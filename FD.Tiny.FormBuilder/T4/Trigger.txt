



 
CREATE OR REPLACE TRIGGER tri_T_API_bir
BEFORE INSERT ON  T_API
FOR EACH ROW
when (new.API_ID IS NULL)
BEGIN
  :new.API_ID := SEQ_T_API.NEXTVAL;
END;
